* npm init -y (terminal)
* .gitignore file (inside file : node_modules)
* npm i -D -g typescript (terminal)
* npm i express (terminal)
* tsc --init
* create src file to save typescript file 
* create dist file to save js file
* change file location (rootDir: .\src) (outDir: ./dist)
* create first file (app.ts & server.ts) --entry point 
* on app.ts file
   * {
      import express from 'express'
      const app: Application = express()
    
      app.get('/', (req :Request, res: Response) => {
         res.send("Hello World!!")
      })
   * }
* on server.ts file 
   * {
      import app fro "./app"
      let server;
      const port = 3000

      const bootstrap = async () => {
         server = app.listen(port, () => {
            console.log(`Example app listening on port ${port}`)
         })
      }
      
      bootstrap()
   * }
* tsc(terminal: to compile express file to js file)
* node ./dist/app.js (terminal: to run file)
